author_counts = []
author_counts_index = -1
current_author = ''
for i in 0...commits.length
  commit = commits[i]
  if commit.author == current_author
    author_counts[author_counts_index][1] += 1
  else
    current_author = commit.author
    author_counts << [commit.author, 1]
    author_counts_index += 1
  end
end
